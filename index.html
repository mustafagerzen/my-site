Q2 – Average of Three Scores (int vs. double)

Üç notun ortalamasını hem int hem double olarak göster, casting örneği.

import java.util.Scanner;
public class Q2_Average {
  public static void main(String[] args){
    Scanner in = new Scanner(System.in);
    System.out.print("Score 1: "); int s1 = in.nextInt();
    System.out.print("Score 2: "); int s2 = in.nextInt();
    System.out.print("Score 3: "); int s3 = in.nextInt();

    int sum = s1 + s2 + s3;
    int avgInt = sum / 3;                 // integer division
    double avgDouble = sum / 3.0;        // double division
    double avgCast = (double) sum / 3;   // explicit cast

    System.out.println("Average (int)   : " + avgInt);
    System.out.println("Average (double): " + avgDouble);
    System.out.println("Average (cast)  : " + avgCast);
  }
}

Q3 – Minutes to Hours:Minutes

Mod ve bölüm kullanımı (%, /): dakika girişini saat:dakika formatına çevir.

import java.util.Scanner;
public class Q3_TimeConvert {
  public static void main(String[] args){

    int hours = minutes / 60;

    System.out.println(minutes + " min = " + hours + "h " + mins + "m");
  }
}

Q4 – Circle Area & Circumference

final sabit ve double ile hesaplama.

import java.util.Scanner;
public class Q4_Circle {
  public static void main(String[] args){
    final double PI = 3.141592653589793;
    Scanner in = new Scanner(System.in);
    System.out.print("Radius: ");
    double r = in.nextDouble();
    double area = PI * r * r;
    double circ = 2 * PI * r;
    System.out.println("Area = " + area);
    System.out.println("Circumference = " + circ);
  }
}

Q9 – Unit Conversion (kg ↔ lb)

final sabit ve çarpma/bölme ile dönüşüm.

import java.util.Scanner;
public class Q9_Units {
  public static void main(String[] args){
    final double KG_PER_LB = 0.45359237; // 1 lb = 0.45359237 kg
    Scanner in = new Scanner(System.in);
    System.out.print("Value: ");
    double val = in.nextDouble();
    System.out.print("Type (1=kg→lb, 2=lb→kg): ");
    int type = in.nextInt();

    double result;
    if (type == 1) result = val / KG_PER_LB;  // kg→lb
    else result = val * KG_PER_LB;            // lb→kg

    System.out.println("Result = " + result);
  }
}

Q10 – Simple Interest

Basit faiz: A = P + P·r·t (yıllık, bileşik değil).

import java.util.Scanner;
public class Q10_SimpleInterest {
  public static void main(String[] args){
    Scanner in = new Scanner(System.in);
    System.out.print("Principal (P): ");
    double P = in.nextDouble();
    System.out.print("Rate (%) (r): ");
    double rPercent = in.nextDouble();
    System.out.print("Time (years, t): ");
    double t = in.nextDouble();

    double r = rPercent / 100.0;
    double A = P + P * r * t;
    System.out.println("Amount A = " + A);
  }
}
