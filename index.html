Q2 – Ice Cream Sales

3 gün boyunca 3 tat için satışları toplayıp ortalamayı hesaplayan kod:

const days = [
  { chocolate: 30, vanilla: 20, strawberry: 25 },
  { chocolate: 25, vanilla: 22, strawberry: 30 },
  { chocolate: 35, vanilla: 18, strawberry: 28 }
];

let total = { chocolate: 0, vanilla: 0, strawberry: 0 };
days.forEach(d => {
  total.chocolate += d.chocolate;
  total.vanilla += d.vanilla;
  total.strawberry += d.strawberry;
});

const avg = {
  chocolate: total.chocolate / days.length,
  vanilla: total.vanilla / days.length,
  strawberry: total.strawberry / days.length
};
console.log(avg);

Q3 – Rocket to the Stars

function rocketMotion(mass, thrust, burnTime) {
  const acceleration = thrust / mass;
  const velocity = acceleration * burnTime;
  return { acceleration, velocity };
}
console.log(rocketMotion(1000, 5000, 10));

Q4 – Newton’s Method

function sqrtApprox(n) {
  let guess = n / 2;
  for (let i = 0; i < 3; i++) {
    guess = (guess + n / guess) / 2;
  }
  return guess;
}
console.log(sqrtApprox(25));

Q5 – Which Day

function whichDay(year, month, day) {
  const d = new Date(year, month - 1, day);
  return d.getDay(); // 0 = Sunday
}
console.log(whichDay(2025, 9, 14));

Q6 – Wind Chill

function windChill(T, V) {
  if (T > 50 || V <= 3) return "N/A";
  return 35.74 + 0.6215*T - 35.75*(V**0.16) + 0.4275*T*(V**0.16);
}
console.log(windChill(30, 10));

Q8 – Triangle Area (Heron)

function heronArea(a, b, c) {
  const s = (a + b + c) / 2;
  return Math.sqrt(s * (s - a) * (s - b) * (s - c));
}
console.log(heronArea(3, 4, 5));

Q10 – Barycenter Calculation

function barycenter(m1, x1, m2, x2) {
  return (m1*x1 + m2*x2) / (m1 + m2);
}
console.log(barycenter(5, 0, 3, 10));
